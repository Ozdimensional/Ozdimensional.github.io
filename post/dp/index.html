1、01背包
#include <bits/stdc++.h>
using namespace std;
int n,m,f[1002],v[1002],w[1002];
int main()
{
cin>>n>>m;
for(int i=1;i<=n;i++)
{
scanf("%d%d",&v[i],&w[i]);
}
for(int i=1;i<=n;i++)
for(int j = m ; j >= v[i];j--)
f[j]=max(f[j],f[j-v[i]]+w[i]);
cout<<f[m];
return 0;
}
