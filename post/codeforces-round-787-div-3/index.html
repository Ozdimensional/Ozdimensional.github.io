}

B. Make It Increasing
贪心a[i-1]>=a[i]的时候a[i-1]/2等于0的时候判断位置即可
#include
using namespace std;
long long a[32];
int main()
{

int t;
scanf("%d",&t);
while(t--)
{

    int n;
    scanf("%d",&n);
    for(int i=1;i<=n;i++)
        scanf("%lld",&a[i]);
    int ans=0;bool flag=true;
    if(a[n]==0&&n!=1)
    {
        flag=false;
    }
    else
    {

        for(int i=n-1;i;i--)
        {
            while(a[i]>=a[i+1])
            {
                a[i]/=2;
                ans++;
            }
            if(a[i]==0&&i!=1)
            {
                flag=false;
                break;
            }
        }
    }
    if(flag)
        cout<<ans<<"\n";
    else
        cout<<"-1\n";

}
return 0;

}
C. Detective Task
像小学的时候做的思维训练题
从后往前第一个1到第一个0之间的所有人都可能偷（包括1和0）
#include <bits/stdc++.h>

using namespace std;
int main()
{
int t;
scanf("%d",&t);
while(t--)
{

    string s;
    cin>>s;
    int k=s.size();
    int p1=0;
    for(int i=k-1;i>=0;i--)
        if(s[i]=='1')
    {
        p1=i;
        break;
    }
    int ans=0;
    for(int i=p1;i<k;i++)
    {
        if(s[i]=='0')
        {
            ans++;
            break;
        }
        else
            ans++;
    }
    cout<<ans<<"\n";

}
return 0;

}

D. Vertical Paths

dfs嗯搜
vector建一棵树
在建立一个vector存输出（题面有更改本来不用统计一共有多少组输出）
#include <bits/stdc++.h>

using namespace std;
const int N=200004;
bool flag[N];
vectoroutp[N];
vectortree[N];
int suma[N];
int ans;
void dfs(int x ,int sum)
{
for(auto i:tree[x])
{
if(!flag[i])
{
if(outp[ans].empty())
sum=0;
sum++;
outp[ans].push_back(i);
flag[i]=true;
if(tree[i].empty())
{
suma[ans]=sum;
ans++;
sum=0;
}
else
{
dfs(i,sum);
}
}
}
}
int main()
{

int t;
scanf("%d",&t);
while(t--)
{
    ans=0;
    int n;
    scanf("%d",&n);
    int x;int start;
    for(int i=0;i<=n;i++)
    {
        tree[i].clear();
        outp[i].clear();
        flag[i]=false;
    }
    for(int i=1;i<=n;i++)
    {

        scanf("%d",&x);
        if(x==i)
        {
            start=x;
        }
        else
            tree[x].push_back(i);

    }
    outp[0].push_back(start);
    flag[start]=true;
    dfs(start,1);
    if(n==1)
        cout<<"1\n1\n1\n\n";
    else{
    cout<<ans<<"\n";
    for(int i=0;i<ans;i++)
    {
        cout<<suma[i]<<"\n";
        for(auto x:outp[i])
            cout<<x<<" ";
        cout<<"\n";
    }
    cout<<"\n";}
}
return 0;

}
