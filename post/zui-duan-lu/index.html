1、朴素dijkstra
#include <bits/stdc++.h>

using namespace std;
int n,m;
int g[510][510];
int dist[510];
bool st[510];
int djkstra()
{
memset(dist,0x3f,sizeof(dist));
dist[1]=0;

for(int i=0;i<n-1;i++)
{
    int t=-1;
    for(int j=1;j<=n;j++)
    {
        if(!st[j]&&(t==-1||dist[t]>dist[j]))
            t=j;
    }
    for(int j=1;j<=n;j++)
        dist[j]=min(dist[j],dist[t]+g[t][j]);
    st[t]=true;
}
    if(dist[n]==0x3f3f3f3f) return -1;
    return dist[n];

}
int main()
{

memset(g,0x3f,sizeof(g));
cin>>n>>m;
while(m--)
{
    int x,y,z;
    cin>>x>>y>>z;
    g[x][y]=min(g[x][y],z);
}
int k=djkstra();
cout<<k<<endl;
return 0;

}
